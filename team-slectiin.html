python
def calculate_fantasy_points(stats, position):
    points = 0

    # General
    points += (stats.minutes_played // 15) * 1
    if stats.card == "black":
        points -= 2
    elif stats.card == "red":
        points -= 5
    if stats.man_of_the_match:
        points += 5

    # Attack
    points += stats.points_scored * 3
    points += stats.goals_scored * 8
    points += stats.assists * 2

    # Defence
    if position in ["GK", "DEF"] and stats.goals_conceded == 0:
        points += 4
    if position in ["GK", "DEF"]:
        points += stats.blocked_shots * 1
        points += stats.turnovers_won * 1
    if position == "GK":
        points -= stats.goals_conceded * 1

    return points
